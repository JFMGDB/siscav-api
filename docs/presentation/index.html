Técnica: Detecção de Bordas (Canny)

Como Funciona

O algoritmo de Canny detecta bordas através de múltiplas etapas:

  1. Suavização: Reduz ruído com filtro Gaussiano
  2. Gradiente: Calcula intensidade e direção das mudanças
  3. Supressão não-máxima: Mantém apenas bordas mais fortes
  4. Histerese: Conecta bordas usando dois thresholds (alto e baixo)

def detect_plates(self, frame):
    # 1. Converter para escala de cinza
    gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
    
    # 2. Equalização de histograma
    gray_eq = cv2.equalizeHist(gray)
    
    # 3. Filtro bilateral (reduz ruído)
    gray_filtered = cv2.bilateralFilter(
        gray_eq, 9, 75, 75
    )
    
    # 4. Detecção de bordas (Canny)
    edges = cv2.Canny(
        gray_filtered, 50, 150
    )
    
    # 5. Operações morfológicas
    kernel = cv2.getStructuringElement(
        cv2.MORPH_RECT, (3, 3)
    )
    edges = cv2.morphologyEx(
        edges, cv2.MORPH_CLOSE, kernel

def read_plate(self, plate_img):
    # 1. Pré-processamento adaptativo
    plate_processed, color = self.preprocess_plate(
        plate_img, vehicle_type
    )
    
    # 2. OCR com EasyOCR
    results = self.reader.readtext(
        plate_processed,
        detail=0,
        allowlist="ABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789"
    )
    
    # 3. Normalização
    text = "".join(results).replace(" ", "").upper()
    normalized = normalize_plate(text)
    
    # 4. Validação formato brasileiro
    is_valid, validated_plate = validate_brazilian_plate(
        normalized
    )
    
    # Formatos aceitos:
    # - Antigo: ABC1234
    # - Mercosul: ABC1D23
    
    return validated_plate if is_valid else None

Implementação - Fluxo Principal

main() - Loop Principal

def main():
    # Inicialização de serviços
    camera = CameraService()
    ocr_service = OCRService()
    plate_detector = PlateDetector()
    api_client = APIClient()
    debouncer = PlateDebouncer()
    
    while True:
        # 1. Capturar frame
        frame = camera.read_frame()
        
        # 2. Detectar placas
        candidates = plate_detector.detect_plates(frame)
        
        # 3. Processar cada candidato
        for x, y, w, h in candidates:
            # Extrair região
            plate_region = frame[y:y+h, x:x+w]
            
            # OCR e validação
            plate_text = ocr_service.read_plate(plate_region)
            
            # Debounce (evitar duplicatas)
            if debouncer.should_process(plate_text):
                # Enviar para API
                response = api_client.send_access_log(
                    plate_text, plate_region
                )
                
                # Feedback visual/sonoro
                draw_detection(frame, x, y, w, h, 
                             plate_text, response)
